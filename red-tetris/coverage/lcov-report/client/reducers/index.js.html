function creatTab() {
  const tab = []
  for(let c = 0; c < 21; c++) {
 
    let tab_tmp = []
    for(let d = 0; d < 10; d++) {
      tab_tmp.push({color: 'white'})
    }
    tab.push(tab_tmp)
  }
  return tab
}
 
const newLine = (color) => {
  const line = []
  for (let i = 0; i < 10; i++) line.push({ color })
  return line
}
 
const putMalusToBoard = (board, malus) => {
  let i = 0
  while (i < malus) {

    board.shift()

}
